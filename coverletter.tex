% CONFIGURATIONS
%-------------------------------------------------------------------------------
% A4 paper size by default, use 'letterpaper' for US letter:
\documentclass[11pt, a4paper]{awesome-cv}

% Configures page margins
\geometry{left=2.5cm, top=1.7cm, right=2.0cm, bottom=1.69cm, footskip=.4cm}

% Specify the location of the included fonts
\fontdir[fonts/]

% Load user-configuration
\input{setup.tex}


%-------------------------------------------------------------------------------
\begin{document}

% Print the header with the given personal informations
% Give the optional argument to change alignment
%  (C: center, L: left, R: right)
\makecvheader[R]

% Print the footer with three optional arguments
%  (<left>, <center>, <right>)
\makecvfooter
  {\today}
  {\makefullname~~~Â·~~~Cover Letter}
  {}

% Print the title with the given letter informations
\makelettertitle


\begin{cvletter}
\input{letter/introduction.tex}
\input{letter/why-this-company.tex}
\input{letter/why-me.tex}
\vspace{1.3em}
\input{letter/closing.tex}
\end{cvletter}


% Print the signature and enclosures with the given letter informations
\makeletterclosing

\end{document}
